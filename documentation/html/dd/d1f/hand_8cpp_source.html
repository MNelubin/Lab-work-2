77std::vector<std::string> Hand::get_unique_card_types() const {
78 std::vector<std::string> types;
79 for (const auto& card : cards) {
80 std::string type = typeid(*card).name();
81 if (std::find(types.begin(), types.end(), type) == types.end()) {
82 types.push_back(type);
83 }
84 }
85 return types;
86}
