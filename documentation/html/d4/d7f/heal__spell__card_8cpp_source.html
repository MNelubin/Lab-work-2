81void Heal_Spell_Card::generate_properties() {
82 // Generate random rarity first
83 Rarity random_rarity = generate_random_rarity();
84 set_rarity(random_rarity);
85
86 std::uniform_int_distribution<int> name_dist(0, heal_names.size() - 1);
87 std::uniform_int_distribution<int> desc_dist(0, heal_descriptions.size() - 1);
88
89 set_name(heal_names[name_dist(rng)]);
90 set_description(heal_descriptions[desc_dist(rng)]);
91 set_element(static_cast<Element>(rng() % 4));
92
93 switch (get_rarity()) {
94 case Rarity::Common:
95 base_heal = 3 + static_cast<int>(rng() % 6);
96 efficiency_deg = 1.0f + static_cast<float>(rng() % 3) * 0.1f;
97 mana_cost = 1 + static_cast<int>(rng() % 2);
98 break;
99 case Rarity::Uncommon:
100 base_heal = 3 + static_cast<int>(rng() % 10);
101 efficiency_deg = 1.2f + static_cast<float>(rng() % 5) * 0.1f;
102 mana_cost = 1 + static_cast<int>(rng() % 3);
103 break;
104 case Rarity::Rare:
105 base_heal = 6 + static_cast<int>(rng() % 15);
106 efficiency_deg = 1.4f + static_cast<float>(rng() % 8) * 0.1f;
107 mana_cost = 2 + static_cast<int>(rng() % 3);
108 break;
109 case Rarity::Epic:
110 base_heal = 10 + static_cast<int>(rng() % 25);
111 efficiency_deg = 1.7f + static_cast<float>(rng() % 12) * 0.1f;
112 mana_cost = 2 + static_cast<int>(rng() % 5);
113 break;
114 }
