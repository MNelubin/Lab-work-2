49void Buff_Spell_Card::generate_properties() {
50 // Generate random rarity first
51 Rarity random_rarity = generate_random_rarity();
52 set_rarity(random_rarity);
53
54 std::uniform_int_distribution<int> name_dist(0, buff_spell_names.size() - 1);
55 std::uniform_int_distribution<int> desc_dist(0, buff_spell_descriptions.size() - 1);
56
57 set_name(buff_spell_names[name_dist(rng)]);
58 set_description(buff_spell_descriptions[desc_dist(rng)]);
59 set_element(static_cast<Element>(rng() % 4));
60
61 switch (get_rarity()) {
62 case Rarity::Common:
63 multiplier = 1.1f + static_cast<float>(rng() % 5) * 0.1f;
64 mana_cost = 1 + static_cast<int>(rng() % 2);
65 break;
66 case Rarity::Uncommon:
67 multiplier = 1.3f + static_cast<float>(rng() % 8) * 0.1f;
68 mana_cost = 1 + static_cast<int>(rng() % 3);
69 break;
70 case Rarity::Rare:
71 multiplier = 1.6f + static_cast<float>(rng() % 11) * 0.1f;
72 mana_cost = 2 + static_cast<int>(rng() % 3);
73 break;
74 case Rarity::Epic:
75 multiplier = 2.0f + static_cast<float>(rng() % 16) * 0.1f;
