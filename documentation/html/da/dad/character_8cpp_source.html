195{
196 // Check stream state
197 if (!os.good())
198 {
199 throw std::ios_base::failure("Stream state invalid before serialization");
200 }
201
202 // Write base fields
203 os.write(reinterpret_cast<const char*>(&xp_to_next_lvl), sizeof(xp_to_next_lvl));
204 os.write(reinterpret_cast<const char*>(&lvl), sizeof(lvl));
205 os.write(reinterpret_cast<const char*>(&xp), sizeof(xp));
206
207 // Write name
208 size_t name_len = name.size();
209 os.write(reinterpret_cast<const char*>(&name_len), sizeof(name_len));
210 os.write(name.c_str(), name_len);
211
212 // Write multipliers
213 os.write(reinterpret_cast<const char*>(&heal_mltpl), sizeof(heal_mltpl));

217 // Write description
218 size_t desc_len = description.size();

233void Character::deserialize(std::istream& is)
234{
235 // Check stream state
236 if (!is.good())
237 {
238 throw std::ios_base::failure("Stream state invalid before deserialization");
239 }
240
241 // Read base fields
242 is.read(reinterpret_cast<char*>(&xp_to_next_lvl), sizeof(xp_to_next_lvl));
243 is.read(reinterpret_cast<char*>(&lvl), sizeof(lvl));
244 is.read(reinterpret_cast<char*>(&xp), sizeof(xp));
245
246 // Read name
247 size_t name_len;
248 is.read(reinterpret_cast<char*>(&name_len), sizeof(name_len));
249 if (name_len > 50)
250 {
251 throw std::ios_base::failure("Corrupted name length (max 50)");
252 }
253 name.resize(name_len);
254 is.read(&name[0], name_len);
255
256 // Read multipliers
257 is.read(reinterpret_cast<char*>(&heal_mltpl), sizeof(heal_mltpl));
258 is.read(reinterpret_cast<char*>(&dmg_mltpl), sizeof(dmg_mltpl));
259 is.read(reinterpret_cast<char*>(&armor_mltpl), sizeof(armor_mltpl));
260
261 // Read description
262 size_t desc_len;
263 is.read(reinterpret_cast<char*>(&desc_len), sizeof(desc_len));
264 if (desc_len > 200)
265 {
266 throw std::ios_base::failure("Corrupted description length (max 200)");
267 }
268 description.resize(desc_len);
269 is.read(&description[0], desc_len);
270
271 // Skip stored ability_uses value
272 int stored_ability_uses;
273 is.read(reinterpret_cast<char*>(&stored_ability_uses), sizeof(stored_ability_uses));
274
275 // Set ability_uses to character's level
276 ability_uses = lvl;
277
278 // Final stream check
279 if (is.fail())
280 {
281 throw std::ios_base::failure("Corrupted character data");
282 }
