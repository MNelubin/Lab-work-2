60void Attack_Spell_Card::generate_properties()

62 // Generate random rarity first

64 set_rarity(random_rarity);
65
66 // Randomly select name, description, and element
67 std::uniform_int_distribution<int> name_dist(0, spell_names.size() - 1);
68 std::uniform_int_distribution<int> desc_dist(0, spell_descriptions.size() - 1);
69
70 set_name(spell_names[name_dist(rng)]);
71 set_description(spell_descriptions[desc_dist(rng)]);
72 set_element(static_cast<Element>(rng() % 4));
73
74 // Generate damage based on rarity
75 switch (get_rarity())
76 {
77 case Rarity::Common:
78 base_damage = 3 + static_cast<int>(rng() % 4);

81 case Rarity::Uncommon:

83 mana_cost = 1 + static_cast<int>(rng() % 3);
