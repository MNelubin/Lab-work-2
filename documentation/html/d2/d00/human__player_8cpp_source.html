49void Human_Player::load_characters(const std::string& filename)
50{
51 std::ifstream file(filename, std::ios::binary);
52 if (!file)
53 {
54 std::cout << "No save data found.\n";
55 return;
56 }
57
58 int num_chars;
59 file.read(reinterpret_cast<char*>(&num_chars), sizeof(num_chars));
60
61
62 for (int i = 0; i < num_chars; ++i)
63 {
64 size_t name_len;
65 file.read(reinterpret_cast<char*>(&name_len), sizeof(name_len));
66 std::string type_name(name_len, '\0');
67 file.read(&type_name[0], name_len);
68
69 std::unique_ptr<Character> character;
70 if (type_name == typeid(Healer_Character).name())
71 {
72 auto healer = std::make_unique<Healer_Character>();
73 healer->deserialize(file);
74 character = std::move(healer);
75 }
76 else if (type_name == typeid(Knight_Character).name())
77 {
78 auto knight = std::make_unique<Knight_Character>();
79 knight->deserialize(file);
80 character = std::move(knight);
81 }
82 else if (type_name == typeid(Tank_Character).name())
83 {
84 auto tank = std::make_unique<Tank_Character>();
85 tank->deserialize(file);
86 character = std::move(tank);
87 }
88 else
89 {
90 std::cerr << "Unknown character type: " << type_name << "\n";
91 continue;
92 }
93
94 characters.push_back(std::move(character));
95 }
96}
