117void Player::eat_card(int index)
118{
119 if (index < 0 || index >= hand.get_amount())
120 {
121 throw std::out_of_range("Invalid card index");
122 }
123
124 // Get the card from hand
125 const Card& card = hand.get_card(index);
126
127 // Get character's HP multiplier
128 float hp_multiplier = get_character().get_heal_multiplier();
129
130 // Calculate rarity value (Common=1, Uncommon=2, Rare=3, Epic=4)
131 int rarity_value = static_cast<int>(card.get_rarity()) + 1;
132
133 // Calculate HP increase with multiplier and convert to integer
134 int hp_increase = static_cast<int>(rarity_value * 10 * hp_multiplier);
135
136 // Update player's HP
137 hp += hp_increase;
138
139 // Remove the card from hand
140 hand.remove_card(index);
141}

143void Player::use_card(int index, Player& target)
144{
145 if (index < 0 || index >= hand.get_amount())
146 {
147 throw std::out_of_range("Invalid card index");
148 }
149
150 // Get reference to the card before removal
151 Card& card = hand.get_card(index);
152 const CardType card_type = card.get_type();
153
154 // Check mana cost
155 int mana_cost = card.get_mana_cost();
156 if (mana < mana_cost)
157 {
158 throw std::runtime_error("Not enough mana to use this card");
159 }
160 mana -= mana_cost;
161
162 // Apply card effect
163 card.use(*this, target);
164
165 // Remove card from hand
166 hand.remove_card(index);
167
168 // Check if card type ends the player's turn
169 if (card_type == CardType::AttackSpell ||
170 card_type == CardType::HealSpell ||
171 card_type == CardType::Beast)
172 {
173 this->end_turn();
174 }

177void Player::take_damage(int damage)
178{
179 if (damage < 0)
180 {
181 throw std::invalid_argument("Damage cannot be negative");
182 }
183
184 int remaining_damage = damage;
185
186 // Apply damage to shields first
187 if (shield_amount > 0)
188 {
189 const int max_shield_absorb = shield_amount * 10;
190 const int absorbed_by_shields = std::min(remaining_damage, max_shield_absorb);
191 remaining_damage -= absorbed_by_shields;
192
193 // Calculate number of shields used (ceiling division)
194 const int shields_used = (absorbed_by_shields + 9) / 10;
195 shield_amount = std::max(shield_amount - shields_used, 0);
196 }
197
198 if (remaining_damage <= 0) return;
199
200 // Apply damage to armor next
201 const int armor_absorb = std::min(remaining_damage, armor);
202 remaining_damage -= armor_absorb;
203 armor = std::max(armor - armor_absorb, 0);
204
205 if (remaining_damage <= 0) return;
206
207 // Apply remaining damage to HP
208 hp = std::max(hp - remaining_damage, 0);
209}
